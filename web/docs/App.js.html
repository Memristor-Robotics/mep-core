        Log.debug(TAG, Config.DEBUG);

            output: process.stdout
        });

        rl.question('What do you think of Node.js? ', (answer) => {
            // TODO: Log the answer in a database
            console.log('Thank you for your valuable feedback:', answer);

            rl.close();
        });

	}
}
